   182→#### Tâche 1.4 : Réécrire POST /api/generate
   183→
   184→Fichier : `src/app/api/generate/route.ts`
   185→
   186→Remplacer l'appel synchrone par la création d'un job BullMQ :
   187→
   188→```typescript
   189→import { NextRequest, NextResponse } from "next/server";
   190→import { z } from "zod";
   191→import { generateQueue } from "@/lib/queue";
   192→import { db } from "@/lib/db";
   193→import { GENRES, MOODS, STYLES, LANGUAGES, AUDIO_FORMATS, MUSICAL_KEYS, TIME_SIGNATURES, STYLE_TO_CAPTION, GenerateRequest } from "@/lib/types";
   194→import { getCurrentUserId } from "@/lib/session";
   195→
   196→const generateSchema = z.object({
   197→  prompt: z.string().trim().min(3).max(500),
   198→  genre: z.enum(GENRES),
   199→  mood: z.enum(MOODS),
   200→  style: z.enum(STYLES),
   201→  voice: z.string().trim().optional(),
   202→  duration: z.number().int().min(10).max(300).optional(),
   203→  language: z.enum(LANGUAGES.map(l => l.code) as [string, ...string[]]).optional(),
   204→  highQuality: z.boolean().optional(),
   205→  customLyrics: z.string().trim().min(20).max(2000).optional(),
   206→  customTitle: z.string().trim().min(1).max(80).optional(),
   207→  audioFormat: z.enum(AUDIO_FORMATS.map(f => f.code) as [string, ...string[]]).optional(),
   208→  bpm: z.number().int().min(30).max(300).optional(),
   209→  keyScale: z.enum(MUSICAL_KEYS as [string, ...string[]]).optional(),
   210→  timeSignature: z.enum(TIME_SIGNATURES.map(t => t.code) as [string, ...string[]]).optional(),
   211→  seed: z.number().int().min(0).max(4294967295).optional(),
   212→});
   213→
   214→export async function POST(req: NextRequest) {
   215→  const userId = await getCurrentUserId();
   216→  if (!userId) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
   217→
   218→  let body: unknown;
   219→  try { body = await req.json(); }
   220→  catch { return NextResponse.json({ error: "Invalid JSON" }, { status: 400 }); }
   221→
   222→  const parsed = generateSchema.safeParse(body);
   223→  if (!parsed.success) {
   224→    return NextResponse.json({ error: parsed.error.issues[0]?.message }, { status: 400 });
   225→  }
   226→
   227→  try {
   228→    // Créer le job dans BullMQ
   229→    const job = await generateQueue.add("generate", {
   230→      userId,
   231→      ...parsed.data,
   232→      caption: buildCaption(parsed.data),
   233→    });
   234→
   235→    // Créer l'enregistrement de suivi en DB
   236→    await db.generationJob.create({
   237→      data: {
   238→        jobId: job.id,
   239→        status: "queued",
   240→        userId,
   241→        prompt: parsed.data.prompt,
   242→        genre: parsed.data.genre,
   243→        mood: parsed.data.mood,
   244→        style: parsed.data.style,
   245→        params: JSON.stringify(parsed.data),
   246→      },
   247→    });
   248→
   249→    return NextResponse.json({ jobId: job.id, status: "queued" }, { status: 202 });
   250→  } catch (err) {
   251→    console.error("generate: failed to queue job", err);
   252→    return NextResponse.json({ error: "Failed to queue generation" }, { status: 500 });
   253→  }
   254→}
   255→
   256→function buildCaption(params: { prompt: string; genre: string; mood: string; style: string }): string {
   257→  const styleHint = STYLE_TO_CAPTION[params.style] ?? "";
   258→  return [params.prompt, params.genre, params.mood.toLowerCase(), styleHint].filter(Boolean).join(", ");
   259→}
   260→```

   319→Fichier : `worker/index.ts`
   320→```typescript
   321→import { Worker } from "bullmq";
   322→import Redis from "ioredis";
   323→import { PrismaClient } from "@prisma/client";
   324→import { S3Client, PutObjectCommand } from "@aws-sdk/client-s3";
   325→import { generateLyrics } from "./handlers/lyrics";
   326→import { generateMusic } from "./handlers/ace-step";
   327→import { generateCover } from "./handlers/cover";
   328→
   329→const connection = new Redis(process.env.REDIS_URL || "redis://localhost:6379");
   330→const db = new PrismaClient();
   331→const s3 = new S3Client({
   332→  region: process.env.S3_REGION || "auto",
   333→  endpoint: process.env.S3_ENDPOINT,
   334→  credentials: {
   335→    accessKeyId: process.env.S3_ACCESS_KEY!,
   336→    secretAccessKey: process.env.S3_SECRET_KEY!,
   337→  },
   338→});
   339→
   340→const BUCKET = process.env.S3_BUCKET || "spotibot";
   341→
   342→async function updateProgress(jobId: string, status: string, progress: number, error?: string) {
   343→  await db.generationJob.update({
   344→    where: { jobId },
   345→    data: { status, progress, error, updatedAt: new Date() },
   346→  });
   347→  // Publier sur Redis pour le WebSocket
   348→  await connection.publish(`job:${jobId}`, JSON.stringify({ status, progress, error }));
   349→}
   350→
   351→async function uploadToS3(key: string, buffer: Buffer, contentType: string): Promise<string> {
   352→  await s3.send(new PutObjectCommand({
   353→    Bucket: BUCKET,
   354→    Key: key,
   355→    Body: buffer,
   356→    ContentType: contentType,
   357→  }));
   358→  return `${process.env.S3_PUBLIC_URL || `https://${BUCKET}.s3.amazonaws.com`}/${key}`;
   359→}
   360→
   361→const worker = new Worker("ace:generate", async (job) => {
   362→  const { userId, caption, customLyrics, customTitle, prompt, genre, mood, style, duration, language, highQuality, audioFormat, bpm, keyScale, timeSignature, seed } = job.data;
   363→
   364→  try {
   365→    // Étape 1 : Lyrics (15%)
   366→    await updateProgress(job.id, "lyrics", 0.15);
   367→    let title: string;
   368→    let lyrics: string;
   369→
   370→    if (customLyrics && customLyrics.trim()) {
   371→      title = customTitle?.trim() || deriveTitleFromLyrics(customLyrics) || "Untitled";
   372→      lyrics = customLyrics.trim();
   373→    } else {
   374→      const out = await generateLyrics({ prompt, genre, mood, style });
   375→      title = out.title;
   376→      lyrics = out.lyrics;
   377→    }
   378→
   379→    // Étape 2 : Audio via ACE-Step (60%)
   380→    await updateProgress(job.id, "audio", 0.35);
   381→    const audioResult = await generateMusic({
   382→      prompt: caption,
   383→      text: lyrics,
   384→      duration: duration || 30,
   385→      language: language || "en",
   386→      thinking: highQuality || false,
   387→      audioFormat: audioFormat || "mp3",
   388→      bpm,
   389→      keyScale,
   390→      timeSignature,
   391→      seed,
   392→    });
   393→
   394→    // Upload audio to S3
   395→    const audioKey = `audio/${job.id}.${audioResult.format}`;
   396→    const mimeMap: Record<string, string> = { mp3: "audio/mpeg", wav: "audio/wav", flac: "audio/flac", opus: "audio/opus", aac: "audio/aac", wav32: "audio/wav" };
   397→    const audioUrl = await uploadToS3(audioKey, audioResult.buffer, mimeMap[audioResult.format] || "audio/mpeg");
   398→
   399→    await updateProgress(job.id, "audio", 0.6);
   400→
   401→    // Étape 3 : Cover en parallèle (80%)
   402→    await updateProgress(job.id, "cover", 0.65);
   403→    const coverResult = await generateCover({ title, genre, mood, prompt });
   404→
   405→    let coverUrl: string | null = null;
   406→    if (coverResult) {
   407→      const coverKey = `covers/${job.id}.png`;
   408→      coverUrl = await uploadToS3(coverKey, coverResult.buffer, "image/png");
   409→    }
   410→
   411→    await updateProgress(job.id, "cover", 0.8);
   412→
   413→    // Étape 4 : Persist en DB (90%)
   414→    const song = await db.song.create({
   415→      data: {
   416→        title,
   417→        prompt,
   418→        lyrics,
   419→        genre,
   420→        mood,
   421→        style,
   422→        voice: language || "en",
   423→        audioUrl,
   424→        audioFormat: audioResult.format,
   425→        durationMs: Math.round((duration || 30) * 1000),
   426→        coverUrl,
   427→        bpm: bpm || null,
   428→        keyScale: keyScale || null,
   429→        timeSig: timeSignature || null,
   430→        seed: audioResult.seedUsed != null ? BigInt(audioResult.seedUsed) : null,
   431→        ownerId: userId,
   432→        status: "ready",
   433→        jobId: job.id,
   434→      },
   435→    });
   436→
   437→    // Mettre à jour le job avec le songId
   438→    await db.generationJob.update({
   439→      where: { jobId: job.id },
   440→      data: { status: "completed", progress: 1.0, songId: song.id },
   441→    });
   442→
   443→    await connection.publish(`job:${job.id}`, JSON.stringify({
   444→      status: "completed",
   445→      progress: 1.0,
   446→      songId: song.id,
   447→    }));
   448→
   449→    return { songId: song.id, title };
   450→  } catch (err) {
   451→    const errorMsg = err instanceof Error ? err.message : String(err);
   452→    await updateProgress(job.id, "failed", 0, errorMsg);
   453→    throw err;
   454→  }
   455→}, { connection, concurrency: 2, limiter: { max: 1, duration: 30_000 } });
   456→
   457→worker.on("completed", (job) => console.log(`Job ${job.id} completed`));
   458→worker.on("failed", (job, err) => console.error(`Job ${job?.id} failed:`, err.message));
   459→
   460→console.log("Worker started, waiting for jobs...");
   461→
   462→function deriveTitleFromLyrics(lyrics: string): string {
   463→  const lines = lyrics.split("\n").map(l => l.trim()).filter(l => l.length > 0 && !l.startsWith("["));
   464→  return lines[0]?.split(/\s+/).slice(0, 6).join(" ").slice(0, 60) || "";
   465→}

   479→// Interface vers l'API ACE-Step auto-hébergée
   480→const ACE_STEP_API = process.env.ACE_STEP_API || "http://localhost:8000";
   481→
   482→export interface AceGenerationParams {
   483→  prompt: string;
   484→  text: string;
   485→  duration: number;
   486→  language: string;
   487→  thinking: boolean;
   488→  audioFormat: string;
   489→  bpm?: number;
   490→  keyScale?: string;
   491→  timeSignature?: string;
   492→  seed?: number;
   493→}
   494→
   495→export interface AceGenerationResult {
   496→  buffer: Buffer;
   497→  format: string;
   498→  seedUsed?: number;
   499→}
   500→
   501→export async function generateMusic(params: AceGenerationParams): Promise<AceGenerationResult> {
   502→  const payload = {
   503→    prompt: params.prompt,
   504→    lyrics: params.text,
   505→    audio_duration: params.duration,
   506→    infer_step: params.thinking ? 60 : 27,
   507→    guidance_scale: 15,
   508→    scheduler_type: "euler",
   509→    cfg_type: "apg",
   510→    omega_scale: 10,
   511→    actual_seeds: params.seed ? [params.seed] : [Math.floor(Math.random() * 4294967295)],
   512→    guidance_interval: 0.5,
   513→    guidance_interval_decay: 0,
   514→    min_guidance_scale: 3,
   515→    use_erg_tag: true,
   516→    use_erg_lyric: true,
   517→    use_erg_diffusion: true,
   518→    oss_steps: [],
   519→    guidance_scale_text: 0,
   520→    guidance_scale_lyric: 0,
   521→  };
   522→
   523→  const res = await fetch(`${ACE_STEP_API}/generate`, {
   524→    method: "POST",
   525→    headers: { "Content-Type": "application/json" },
   526→    body: JSON.stringify(payload),
   527→    signal: AbortSignal.timeout(300_000),
   528→  });
   529→
   530→  if (!res.ok) {
   531→    const err = await res.text();
   532→    throw new Error(`ACE-Step API error (${res.status}): ${err}`);
   533→  }
   534→
   535→  const result = await res.json();
   536→  // Télécharger le fichier audio généré
   537→  const audioRes = await fetch(`${ACE_STEP_API}/outputs/${result.output_path}`);
   538→  if (!audioRes.ok) throw new Error("Failed to download generated audio");

  1092→#### Tâche 4.2 : API FastAPI ACE-Step modifiée
  1093→
  1094→Fichier : `docker/ace-step/infer-api-mod.py`
  1095→```python
  1096→"""
  1097→API FastAPI asynchrone pour ACE-Step avec file d'attente locale et webhook.
  1098→"""
  1099→from fastapi import FastAPI, HTTPException, UploadFile, File, Form
  1100→from pydantic import BaseModel
  1101→from typing import Optional
  1102→import os, uuid, asyncio
  1103→from acestep.pipeline_ace_step import ACEStepPipeline
  1104→from contextlib import asynccontextmanager
  1105→
  1106→pipeline: Optional[ACEStepPipeline] = None
  1107→
  1108→@asynccontextmanager
  1109→async def lifespan(app: FastAPI):
  1110→    global pipeline
  1111→    pipeline = ACEStepPipeline(
  1112→        checkpoint_dir=os.environ.get("CHECKPOINT_PATH", ""),
  1113→        dtype="bfloat16" if os.environ.get("BF16", "true") == "true" else "float32",
  1114→        torch_compile=True,
  1115→        cpu_offload=True,
  1116→        overlapped_decode=True,
  1117→    )
  1118→    yield
  1119→
  1120→app = FastAPI(title="ACE-Step API", lifespan=lifespan)
  1121→
  1122→class GenerateRequest(BaseModel):
  1123→    prompt: str
  1124→    lyrics: str = ""
  1125→    audio_duration: float = 30.0
  1126→    infer_step: int = 27
  1127→    guidance_scale: float = 15.0
  1128→    scheduler_type: str = "euler"
  1129→    cfg_type: str = "apg"
  1130→    omega_scale: float = 10.0
  1131→    actual_seeds: list[int] = []
  1132→    guidance_interval: float = 0.5
  1133→    guidance_interval_decay: float = 0.0
  1134→    min_guidance_scale: float = 3.0
  1135→    use_erg_tag: bool = True
  1136→    use_erg_lyric: bool = True
  1137→    use_erg_diffusion: bool = True
  1138→    oss_steps: list[int] = []
  1139→    guidance_scale_text: float = 0.0
  1140→    guidance_scale_lyric: float = 0.0
  1141→
  1142→@app.post("/generate")
  1143→async def generate(req: GenerateRequest):
  1144→    global pipeline
  1145→    if not pipeline:
  1146→        raise HTTPException(503, "Pipeline not initialized")
  1147→    
  1148→    output_path = f"/app/outputs/{uuid.uuid4().hex}.wav"
  1149→    seeds = req.actual_seeds or [os.urandom(4)]
  1150→    
  1151→    try:
  1152→        # Exécution synchrone mais dans un thread pool
  1153→        loop = asyncio.get_event_loop()
  1154→        await loop.run_in_executor(None, lambda: pipeline(
  1155→            audio_duration=req.audio_duration,
  1156→            prompt=req.prompt,
  1157→            lyrics=req.lyrics,
  1158→            infer_step=req.infer_step,
  1159→            guidance_scale=req.guidance_scale,
  1160→            scheduler_type=req.scheduler_type,
  1161→            cfg_type=req.cfg_type,
  1162→            omega_scale=req.omega_scale,
  1163→            manual_seeds=", ".join(map(str, seeds)),
  1164→            guidance_interval=req.guidance_interval,
  1165→            guidance_interval_decay=req.guidance_interval_decay,
  1166→            min_guidance_scale=req.min_guidance_scale,
  1167→            use_erg_tag=req.use_erg_tag,
  1168→            use_erg_lyric=req.use_erg_lyric,
  1169→            use_erg_diffusion=req.use_erg_diffusion,
  1170→            oss_steps=", ".join(map(str, req.oss_steps)),
  1171→            guidance_scale_text=req.guidance_scale_text,
  1172→            guidance_scale_lyric=req.guidance_scale_lyric,
  1173→            save_path=output_path,
  1174→        ))
  1175→        return {"status": "success", "output_path": output_path, "seed_used": seeds[0]}
  1176→    except Exception as e:
  1177→        raise HTTPException(500, str(e))
  1178→
  1179→@app.get("/health")
  1180→async def health():
  1181→    return {"status": "healthy", "pipeline_loaded": pipeline is not None}
  1182→```
